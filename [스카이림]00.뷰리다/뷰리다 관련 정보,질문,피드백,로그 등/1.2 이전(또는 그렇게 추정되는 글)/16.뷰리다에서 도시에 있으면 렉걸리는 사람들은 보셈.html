[General]

#If true, update Player's normalmap. This overrides the condition's value

PlayerEnable = true


#If true, update NPC's normalmap. This overrides the condition's value

NPCEnable = false


#If true, update male normalmap. This overrides the condition's value

MaleEnable = false


#If true, update female normalmap. This overrides the condition's value

FemaleEnable = true


#If true, update head normalmap. This overrides the condition's value
